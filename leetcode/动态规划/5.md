class Solution {
public:
    string longestPalindrome(string s) {
        int len = s.size();
        if(len==0 || len==1){
            return s;
        }
        vector<vector<bool>>dp(len, vector<bool>(len, false));
        int maxLen=1, start=0;
        dp[0][0] = true;
        for(int r=1; r<len; r++){
            for(int l=0; l<r; l++){
                if(s[l]==s[r] && (r-l<=2 || dp[l+1][r-1])){
                    dp[l][r] = true;
                    if(r-l+1 > maxLen){
                        maxLen = r-l+1;
                        start = l;
                    }
                }
            }
        }
        return s.substr(start, maxLen);
    }
};
